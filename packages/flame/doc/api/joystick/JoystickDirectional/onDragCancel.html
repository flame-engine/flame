@override
bool onDragCancel(int pointerId) {
  _dragging = false;
  _dragPosition = _backgroundRect.center.toVector2();
  joystickController.joystickChangeDirectional(JoystickDirectionalEvent(
    directional: JoystickMoveDirectional.idle,
  ));
  return true;
}
