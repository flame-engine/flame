static Future<SpriteAnimation> load(
  String src,
  SpriteAnimationData data, {
  Images images,
}) async {
  final _images = images ?? Flame.images;
  final image = await _images.load(src);
  return SpriteAnimation.fromFrameData(image, data);
}
