void render(Canvas canvas, {Vector2 position}) {
  canvas.save();
  if (position != null) {
    canvas.translateVector(position);
  }
  canvas.clipRect(_clipRect);
  layers.forEach((layer) {
    canvas.save();
    layer.render(canvas);
    canvas.restore();
  });
  canvas.restore();
}
